    \pause
    Unstaged fájlok staged állapotba helyezése:
\begin{verbatim}
~/myproject$ git add file.txt
\end{verbatim}

    \pause
    Staged fájlok unstaged állapotba helyezése:
\begin{verbatim}
~/myproject$ git rm --cached file.txt
rm 'file.txt'
\end{verbatim}
\end{frame}
