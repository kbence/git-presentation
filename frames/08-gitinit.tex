
\begin{frame}[fragile]{Git repository inicializálása}
    \pause
    Inicializálás\small
\begin{verbatim}
~/myproject$ git init
Initialized empty Git repository in ~/myproject/.git/
\end{verbatim}
    \normalsize
    
    \pause
    "Bare" repository inicializálás\small
\begin{verbatim}
~/myproject$ git init --bare
\end{verbatim}
    \normalsize
\end{frame}
